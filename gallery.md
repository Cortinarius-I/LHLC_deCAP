<script>
(function() {
  var lb = document.getElementById('lightbox');
  var lbImg = lb.querySelector('.lightbox-img');
  var lbCap = lb.querySelector('.lightbox-caption');
  var lbCount = lb.querySelector('.lightbox-counter');
  var lbPrev = lb.querySelector('.lightbox-prev');
  var lbNext = lb.querySelector('.lightbox-next');

  var sections = [];
  document.querySelectorAll('.gallery-section').forEach(function(sec) {
    var imgs = [];
    sec.querySelectorAll('.gallery-link').forEach(function(link) {
      imgs.push({ src: link.href, caption: link.dataset.caption });
    });
    sections.push(imgs);
  });

  var curSection = 0;
  var curIndex = 0;

  function openLb(si, ii) {
    curSection = si;
    curIndex = ii;
    showCurrent();
    lb.classList.add('active');
    document.body.style.overflow = 'hidden';
  }

  function showCurrent() {
    var sec = sections[curSection];
    var entry = sec[curIndex];
    lbImg.src = entry.src;
    lbImg.alt = entry.caption;
    lbCap.textContent = entry.caption;
    lbCount.textContent = (curIndex + 1) + '\u00a0/\u00a0' + sec.length;
    lbPrev.style.visibility = curIndex > 0 ? 'visible' : 'hidden';
    lbNext.style.visibility = curIndex < sec.length - 1 ? 'visible' : 'hidden';
  }

  function navigate(delta) {
    var sec = sections[curSection];
    var n = curIndex + delta;
    if (n >= 0 && n < sec.length) { curIndex = n; showCurrent(); }
  }

  function closeLb() {
    lb.classList.remove('active');
    document.body.style.overflow = '';
    lbImg.src = '';
  }

  document.querySelectorAll('.gallery-link').forEach(function(link) {
    link.addEventListener('click', function(e) {
      e.preventDefault();
      openLb(parseInt(this.dataset.section), parseInt(this.dataset.index));
    });
  });

  lbPrev.addEventListener('click', function() { navigate(-1); });
  lbNext.addEventListener('click', function() { navigate(1); });
  lb.querySelector('.lightbox-close').addEventListener('click', closeLb);
  lb.addEventListener('click', function(e) { if (e.target === lb) closeLb(); });

  document.addEventListener('keydown', function(e) {
    if (!lb.classList.contains('active')) return;
    if (e.key === 'Escape') closeLb();
    if (e.key === 'ArrowLeft')  navigate(-1);
    if (e.key === 'ArrowRight') navigate(1);
  });
})();
</script>
